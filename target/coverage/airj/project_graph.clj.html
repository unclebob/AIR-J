001  (ns airj.project-graph

005  (defn direct-airj-imports
006    [source]
007    (->> source
008         parser/parse-module
009         normalizer/normalize-module
010         :imports
011         (filter #(= :airj-import (:op %)))
012         (map :module)))

014  (defn- module-source
015    [sources module-name]
016    (or (get sources module-name)
017        (throw (ex-info "Missing project module source."
018                        {:module module-name}))))
019  
020  (defn- fail-on-cycle!
021    [module-name path]
022    (let [cycle-start (.indexOf path module-name)
023          cycle-path (conj (subvec path cycle-start) module-name)]
024      (throw (ex-info "Project import cycle."
025                      {:cycle cycle-path}))))
026  
027  (defn- visit-module
028    [sources module-name visited path reachable]
029    (when (some #{module-name} path)
030      (fail-on-cycle! module-name path))
031    (if (visited module-name)
032      [visited reachable]
033      (let [source (module-source sources module-name)
034            next-path (conj path module-name)
035            imports (direct-airj-imports source)
036            next-visited (conj visited module-name)
037            next-reachable (assoc reachable module-name source)]
038        (reduce (fn [[current-visited current-reachable] imported-module]
039                  (visit-module sources
040                                imported-module
041                                current-visited
042                                next-path
043                                current-reachable))
044                [next-visited next-reachable]
045                imports))))
046  
047  (defn reachable-source-map
048    [sources root-module-name]
049    (second (visit-module sources root-module-name #{} [] {})))
