001  (ns airj.project-sources
002    (:require [airj.interface-sources :as interface-sources]
003              [airj.normalizer :as normalizer]
004              [airj.parser :as parser]
005              [airj.project-graph :as project-graph]))
006  
007  (defn- parsed-module-name
008    [source]
009    (:name (normalizer/normalize-module (parser/parse-module source))))
010  
011  (defn source-map
012    [module-sources]
013    (into {}
014          (map (fn [[module-name source]]
015                 [(or module-name (parsed-module-name source)) source]))
016          module-sources))
017  
018  (defn root-source
019    [module-sources root-module-name]
020    (let [sources (source-map module-sources)]
021      (or (get sources root-module-name)
022          (throw (ex-info "Missing project root module source."
023                          {:module root-module-name})))))
024  
025  (defn reachable-source-map
026    [module-sources root-module-name]
027    (project-graph/reachable-source-map (source-map module-sources) root-module-name))
028  
029  (defn imported-interfaces
030    [module-sources root-module-name]
031    (let [sources (reachable-source-map module-sources root-module-name)]
032      (-> sources
033          (dissoc root-module-name)
034          interface-sources/sources->interfaces)))
035  
036  (defn compiler-options
037    [module-sources root-module-name]
038    {:interfaces (imported-interfaces module-sources root-module-name)})
