001  (ns airj.jvm-class-writer
002    (:import (clojure.asm ClassWriter)))

004  (defn- class-hierarchy
005    [module]
006    (merge {(:internal-name module) "java/lang/Object"}
007           (into {}
008                 (map (fn [record]
009                        [(:class-name record) "java/lang/Object"]))
010                 (:records module))
011           (into {}
012                 (mapcat (fn [union]
013                           (concat [[(:base-class union) "java/lang/Object"]]
014                                   (map (fn [variant]
015                                          [(:class-name variant) (:base-class union)])
016                                        (:variants union))))
017                         (:unions module)))))
018  
019  (defn- ancestor-chain
020    [hierarchy class-name]
021    (loop [current class-name
022           ancestors []]
023      (if current
024        (recur (get hierarchy current)
025               (conj ancestors current))
026        ancestors)))
027  
028  (defn- common-super-class
029    [hierarchy left right]
030    (cond
031      (= left right) left
032      (= left "java/lang/Object") left
033      (= right "java/lang/Object") right
034      :else (let [left-ancestors (set (ancestor-chain hierarchy left))]
035              (or (some left-ancestors (ancestor-chain hierarchy right))
036                  "java/lang/Object"))))
037  
038  (defn class-writer
039    [module]
040    (let [hierarchy (class-hierarchy module)]
041      (proxy [ClassWriter] [(+ ClassWriter/COMPUTE_FRAMES
042                               ClassWriter/COMPUTE_MAXS)]
043        (getCommonSuperClass [left right]
044          (common-super-class hierarchy left right)))))
