001  ;; mutation-tested: 2026-03-11
002  (ns airj.java-effects)
003  
004  (def ^:private pure-static-members
005    #{['java.lang.Math 'abs [Integer/TYPE] Integer/TYPE]})
006  
007  (def ^:private pure-instance-members
008    #{['length [] Integer/TYPE]})
009  
010  (defn- pure-static?
011    [member]
012    (contains? pure-static-members
013               [(:class-name member)
014                (:member-id member)
015                (:parameter-types member)
016                (:return-type member)]))
017  
018  (defn- pure-instance?
019    [member]
020    (contains? pure-instance-members
021               [(:member-id member)
022                (:parameter-types member)
023                (:return-type member)]))
024  
025  (defn- call-effects
026    [member]
027    (if (pure-instance? member)
028      #{}
029      #{'Foreign.Throw}))
030  
031  (defn- static-call-effects
032    [member]
033    (if (pure-static? member)
034      #{}
035      #{'Foreign.Throw}))
036  
037  (defn- field-effects
038    [member]
039    (if (:writable? member)
040      #{'State.Write}
041      #{}))
042  
043  (defn- method-effects
044    [member]
045    (if (:static? member)
046      (static-call-effects member)
047      (call-effects member)))
048  
049  (defn member-effects
050    [member]
051    (case (:kind member)
052      :constructor #{'Foreign.Throw}
053      :method (method-effects member)
054      :field (field-effects member)
055      #{}))
056  
057  (defn- write-field-effects
058    [member]
059    (member-effects (assoc member :writable? true)))
060  
061  (defn interop-effects
062    [expr member]
063    (case (:op expr)
064      :java-new #{'Foreign.Throw}
065      :java-call (member-effects member)
066      :java-static-call (member-effects member)
067      :java-get-field #{}
068      :java-set-field (write-field-effects member)
069      #{}))
