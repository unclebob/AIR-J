010  (defn lower-literal
011    [expr]
012    (cond
013      (integer? expr) {:op :jvm-int

019      (true? expr) {:op :jvm-boolean

078  (defn- lower-call
079    [expr ctx]
080    (let [callee (:callee expr)
081          local-name (local-callee-name callee)]
082      (cond
083        (= :lambda (:op callee))
084        (closures/lower-inline-lambda-call callee
085                                           (:args expr)
086                                           ctx
087                                           types/bind-local
088                                           lower-expr
089                                           types/lower-type)
090  
091        (types/local-lambda ctx local-name)
092        (closures/lower-inline-lambda-call (types/local-lambda ctx local-name)
093                                           (:args expr)
094                                           ctx
095                                           types/bind-local
096                                           lower-expr
097                                           types/lower-type)
098  
099        (types/fn-decl ctx local-name)
100        (lower-static-call (types/fn-decl ctx local-name) (:args expr) ctx)
101  
102        (= :fn-type (:op (types/infer-type callee ctx)))
103        (lower-closure-call callee (:args expr) (types/infer-type callee ctx) ctx)
104  
105        :else
106        (types/fail! "Unknown lowered call target."
107                     {:callee callee}))))

155                    (let [binding-type (types/infer-type (:expr binding) current-ctx)

164                        [(conj acc {:name (:name binding)
