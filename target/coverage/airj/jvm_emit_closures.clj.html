001  (ns airj.jvm-emit-closures
002    (:import (clojure.asm ClassWriter MethodVisitor Opcodes)))

004  (defn emit-closure-new
005    [^MethodVisitor mv expr env {:keys [emit-expr constructor-descriptor]}]
006    (.visitTypeInsn mv Opcodes/NEW (:class-name expr))
007    (.visitInsn mv Opcodes/DUP)
008    (doseq [arg (:args expr)]
009      (emit-expr mv arg env))
010    (.visitMethodInsn mv
011                      Opcodes/INVOKESPECIAL
012                      (:class-name expr)
013                      "<init>"
014                      (constructor-descriptor (:args expr))
015                      false))
016  
017  (defn emit-closure-call
018    [^MethodVisitor mv expr env {:keys [emit-expr descriptor]}]
019    (emit-expr mv (:callee expr) env)
020    (doseq [arg (:args expr)]
021      (emit-expr mv arg env))
022    (.visitMethodInsn mv
023                      Opcodes/INVOKEINTERFACE
024                      (:interface-name expr)
025                      (:method-name expr)
026                      (str "("
027                           (apply str (map descriptor (:parameter-types expr)))
028                           ")"
029                           (descriptor (:return-type expr)))
030                      true))

032  (defn emit-closure-interface-bytes
033    [^ClassWriter cw interface method-descriptor]
034    (.visit cw
035            Opcodes/V1_8
036            (+ Opcodes/ACC_PUBLIC Opcodes/ACC_ABSTRACT Opcodes/ACC_INTERFACE)
037            (:class-name interface)
038            nil
039            "java/lang/Object"
040            nil)
041    (.visitMethod cw
042                  (+ Opcodes/ACC_PUBLIC Opcodes/ACC_ABSTRACT)
043                  (:method-name interface)
044                  (method-descriptor (mapv (fn [jvm-type]
045                                             {:jvm-type jvm-type})
046                                           (:params interface))
047                                     (:return-type interface))
048                  nil
049                  nil)
050    (.visitEnd cw)
051    (.toByteArray cw))

053  (defn emit-closure-class-bytes
054    [^ClassWriter cw closure
055     {:keys [descriptor init-descriptor field-slot load-opcode store-opcode
056             method-descriptor instance-local-slots bind-typed-slots emit-expr
057             return-opcode]}]
058    (let [interfaces (into-array String [(:interface-name closure)])]
059      (.visit cw
060              Opcodes/V1_8
061              (+ Opcodes/ACC_PUBLIC Opcodes/ACC_SUPER)
062              (:class-name closure)
063              nil
064              "java/lang/Object"
065              interfaces)
066      (doseq [capture (:captures closure)]
067        (.visitField cw
068                     (+ Opcodes/ACC_PRIVATE Opcodes/ACC_FINAL)
069                     (str (:name capture))
070                     (descriptor (:jvm-type capture))
071                     nil
072                     nil))
073      (let [mv (.visitMethod cw
074                             Opcodes/ACC_PUBLIC
075                             "<init>"
076                             (init-descriptor (:captures closure))
077                             nil
078                             nil)]
079        (.visitCode mv)
080        (.visitVarInsn mv Opcodes/ALOAD 0)
081        (.visitMethodInsn mv Opcodes/INVOKESPECIAL "java/lang/Object" "<init>" "()V" false)
082        (doseq [[slot capture] (field-slot (:captures closure))]
083          (.visitVarInsn mv Opcodes/ALOAD 0)
084          (.visitVarInsn mv (load-opcode (:jvm-type capture)) slot)
085          (.visitFieldInsn mv
086                           Opcodes/PUTFIELD
087                           (:class-name closure)
088                           (str (:name capture))
089                           (descriptor (:jvm-type capture))))
090        (.visitInsn mv Opcodes/RETURN)
091        (.visitMaxs mv 0 0)
092        (.visitEnd mv))
093      (let [mv (.visitMethod cw
094                             Opcodes/ACC_PUBLIC
095                             (:method-name closure)
096                             (method-descriptor (:params closure) (:return-type closure))
097                             nil
098                             nil)
099            env (bind-typed-slots (instance-local-slots (:params closure))
100                                  (:captures closure)
101                                  :jvm-type)]
102        (.visitCode mv)
103        (doseq [capture (:captures closure)]
104          (let [{:keys [slot jvm-type]} (get-in env [:slots (:name capture)])]
105            (.visitVarInsn mv Opcodes/ALOAD 0)
106            (.visitFieldInsn mv
107                             Opcodes/GETFIELD
108                             (:class-name closure)
109                             (str (:name capture))
110                             (descriptor (:jvm-type capture)))
111            (.visitVarInsn mv (store-opcode jvm-type) slot)))
112        (emit-expr mv (:body closure) env)
113        (.visitInsn mv (return-opcode (:return-type closure)))
114        (.visitMaxs mv 0 0)
115        (.visitEnd mv))
116      (.visitEnd cw)
117      (.toByteArray cw)))
