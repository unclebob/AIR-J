002  (ns airj.java-types)
003  
004  (def ^:private primitive-types
005    {'Bool Boolean/TYPE
006     'Int Integer/TYPE
007     'Unit Void/TYPE})
008  
009  (defn load-class
010    [class-name]
011    (Class/forName (str class-name)))
012  
013  (defn java-type-expr?
014    [type-expr]
015    (and (seq? type-expr)
016         (= "Java" (name (first type-expr)))))
017  
018  (defn resolve-type
019    [type-expr]
020    (cond
021      (contains? primitive-types type-expr) (get primitive-types type-expr)
022      (= 'String type-expr) java.lang.String
023      (java-type-expr? type-expr) (load-class (second type-expr))
024      :else nil))
025  
026  (defn assignable-type-expr?
027    [expected-type actual-type]
028    (let [expected (resolve-type expected-type)
029          actual (resolve-type actual-type)]
030      (cond
031        (or (nil? expected) (nil? actual)) (= expected-type actual-type)
032        (or (.isPrimitive expected) (.isPrimitive actual)) (= expected actual)
033        :else (.isAssignableFrom expected actual))))
