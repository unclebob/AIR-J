014                    (when-not interface

021                                  (fail! "Unresolved imported symbol."
022                                         {:module (:name module)
023                                          :imported-module (:module import)
024                                          :symbol symbol}))
025                                (assoc current symbol
026                                       {:module (:module import)
027                                        :decl decl})))
028                            acc
029                            (:symbols import)))))
