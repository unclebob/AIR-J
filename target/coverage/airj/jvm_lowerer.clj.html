032     :base-class (types/nested-class-name (:module-name ctx) (:name decl))

035  (defn- lower-decls

041  (defn lower-module
042    [module]
043    (let [decls (types/decl-map module)
044          ctx {:module-name (:name module)
045               :decls decls
046               :imported-decls (imported-interfaces/imported-decls module)
047               :closure-counter (atom 0)
048               :closures (atom [])
049               :closure-interfaces (atom {})}
050          lower-method* (fn [decl]
051                          (let [method-ctx (assoc ctx
052                                             :locals (into {}
053                                                           (map (fn [param]
054                                                                  [(:name param) (:type param)]))
055                                                           (:params decl)))]
056                            {:name (:name decl)
057                             :owner (types/internal-name (:module-name ctx))
058                             :params (mapv #(expr/lower-param % method-ctx) (:params decl))
059                             :return-type (types/lower-type (:return-type decl) method-ctx)
060                             :effects (vec (:effects decl))
061                             :body (expr/lower-expr (:body decl) method-ctx)}))]
062      (cond-> {:op :jvm-module
063               :module-name (:name module)
064               :internal-name (types/internal-name (:name module))
065               :exports (vec (:exports module))
066               :records (lower-decls (:decls module) :data #(lower-record ctx %))
067               :enums (lower-decls (:decls module) :enum #(lower-enum ctx %))
068               :unions (lower-decls (:decls module) :union #(lower-union ctx %))
069               :methods (lower-decls (:decls module) :fn lower-method*)}
070        (seq @(:closure-interfaces ctx))
071        (assoc :closure-interfaces (->> @(:closure-interfaces ctx) vals (sort-by :class-name) vec))
072        (seq @(:closures ctx))
073        (assoc :closures (vec @(:closures ctx))))))
