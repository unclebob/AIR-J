001  (ns airj.java-resolver
002    (:require [airj.expr-walker :as expr-walker]
003              [airj.java-members :as java-members]
004              [airj.java-types :as java-types]

007  (defn- fail!
008    [message data]
009    (throw (ex-info message data)))
010  
011  (defn- decl-map
012    [module]
013    (into {} (map (juxt :name identity) (:decls module))))
014  
015  (defn- imported-java-classes
016    [module]
017    (->> (:imports module)
018         (filter #(= :java-import (:op %)))
019         (map :class-name)
020         set))
021  
022  (defn- require-java-import!
023    [class-name imports]
024    (when-not (contains? imports class-name)
025      (fail! "Missing Java import."
026             {:class-name class-name})))
027  
028  (defn- load-class
029    [class-name]
030    (try
031      (java-types/load-class class-name)
032      (catch ClassNotFoundException _
033        (fail! "Unknown Java class."
034               {:class-name class-name}))))
035  
036  (defn resolve-type
037    [type-expr]
038    (or (java-types/resolve-type type-expr)
039        (fail! "Unsupported Java type mapping."
040               {:type type-expr})))
041  
042  (defn- java-class-type?
043    [type-expr]
044    (java-types/java-type-expr? type-expr))
045  
046  (defn- target-class
047    [expr ctx decls imports]
048    (let [target-type (type-checker/infer-expr-type expr ctx decls)]
049      (when-not (java-class-type? target-type)
050        (fail! "Expected Java receiver type."
051               {:type target-type}))
052      (require-java-import! (second target-type) imports)
053      (resolve-type target-type)))
054  
055  (defn- validate-java-new
056    [expr ctx decls imports]
057    (require-java-import! (:class-name expr) imports)
058    (load-class (:class-name expr))
059    (let [arg-types (mapv #(resolve-type (type-checker/infer-expr-type % ctx decls))
060                          (:args expr))]
061      (when-not (java-members/resolve-constructor (:class-name expr) arg-types)
062        (fail! "Unknown Java constructor."
063               {:class-name (:class-name expr)
064                :arg-types arg-types}))))
065  
066  (defn- validate-java-call
067    [expr ctx decls imports]
068    (let [receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
069          klass (target-class (:target expr) ctx decls imports)]
070      (when-not (java-members/resolve-instance-method receiver-type
071                                                      (:member-id expr)
072                                                      (:signature expr))
073        (fail! "Unknown Java method."
074               {:class-name (.getName klass)
075                :member-id (:member-id expr)
076                :signature (:signature expr)}))))
077  
078  (defn- validate-java-static-call
079    [expr _ctx _decls imports]
080    (require-java-import! (:class-name expr) imports)
081    (when-not (java-members/resolve-static-method (:class-name expr)
082                                                  (:member-id expr)
083                                                  (:signature expr))
084      (let [klass (load-class (:class-name expr))]
085        (fail! "Unknown Java method."
086               {:class-name (:class-name expr)
087                :member-id (:member-id expr)
088                :signature (:signature expr)}))))
089  
090  (defn- validate-java-get-field
091    [expr ctx decls imports]
092    (let [receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
093          klass (target-class (:target expr) ctx decls imports)]
094      (when-not (java-members/resolve-instance-field receiver-type
095                                                     (:field-name expr)
096                                                     (:field-type expr))
097        (fail! "Unknown Java field."
098               {:class-name (.getName klass)
099                :field-name (:field-name expr)
100                :field-type (:field-type expr)}))))
101  
102  (defn- validate-java-set-field
103    [expr ctx decls imports]
104    (let [receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
105          klass (target-class (:target expr) ctx decls imports)]
106      (when-not (java-members/resolve-instance-field receiver-type
107                                                     (:field-name expr)
108                                                     (:field-type expr))
109        (fail! "Unknown Java field."
110               {:class-name (.getName klass)
111                :field-name (:field-name expr)
112                :field-type (:field-type expr)}))))

114  (defn- handlers
115    [decls imports]
116    {:java-new (fn [expr ctx walk]
117                 (doseq [arg (:args expr)]
118                   (walk arg ctx))
119                 (validate-java-new expr ctx decls imports))
120     :java-call (fn [expr ctx walk]
121                  (walk (:target expr) ctx)
122                  (doseq [arg (:args expr)]
123                    (walk arg ctx))
124                  (validate-java-call expr ctx decls imports))
125     :java-static-call (fn [expr ctx walk]
126                         (doseq [arg (:args expr)]
127                           (walk arg ctx))
128                         (validate-java-static-call expr ctx decls imports))
129     :java-get-field (fn [expr ctx walk]
130                       (walk (:target expr) ctx)
131                       (validate-java-get-field expr ctx decls imports))
132     :java-set-field (fn [expr ctx walk]
133                       (walk (:target expr) ctx)
134                       (walk (:expr expr) ctx)
135                       (validate-java-set-field expr ctx decls imports))
136     :call (fn [expr ctx walk]
137             (walk (:callee expr) ctx)
138             (doseq [arg (:args expr)]
139               (walk arg ctx)))
140     :construct (fn [expr ctx walk]
141                  (doseq [arg (:args expr)]
142                    (walk arg ctx)))
143     :variant (fn [expr ctx walk]
144                (doseq [arg (:args expr)]
145                  (walk arg ctx)))
146     :record-get (fn [expr ctx walk]
147                   (walk (:target expr) ctx))
148     :if (fn [expr ctx walk]
149           (walk (:test expr) ctx)
150           (walk (:then expr) ctx)
151           (walk (:else expr) ctx))
152     :match (fn [expr ctx walk]
153              (walk (:target expr) ctx)
154              (doseq [case (:cases expr)]
155                (walk (:body case) ctx)))
156     :let (fn [expr ctx walk]
157            (doseq [binding (:bindings expr)]
158              (walk (:expr binding) ctx))
159            (walk (:body expr) ctx))
160     :seq (fn [expr ctx walk]
161            (doseq [part (:exprs expr)]
162              (walk part ctx)))
163     :lambda (fn [expr ctx walk]
164               (walk (:body expr) ctx))
165     :try (fn [expr ctx walk]
166            (walk (:body expr) ctx)
167            (doseq [catch (:catches expr)]
168              (walk (:body catch) ctx))
169            (when-let [finally-expr (:finally expr)]
170              (walk finally-expr ctx)))
171     :var (fn [expr ctx walk]
172            (walk (:init expr) ctx))
173     :set (fn [expr ctx walk]
174            (walk (:expr expr) ctx))
175     :loop (fn [expr ctx walk]
176             (doseq [binding (:bindings expr)]
177               (walk (:expr binding) ctx))
178             (walk (:body expr) ctx))
179     :recur (fn [expr ctx walk]
180              (doseq [arg (:args expr)]
181                (walk arg ctx)))
182     :raise (fn [expr ctx walk]
183              (walk (:expr expr) ctx))})

185  (defn- fn-ctx
186    [decl decls]
187    {:locals (merge
188              (into {}
189                    (map (fn [param]
190                           [(:name param) (:type param)])
191                         (:params decl)))
192              (into {}
193                    (map (fn [fn-decl]
194                           [(:name fn-decl)
195                            {:op :fn-type
196                             :params (mapv :type (:params fn-decl))
197                             :return-type (:return-type fn-decl)
198                             :effects (vec (:effects fn-decl))}])
199                         (filter #(= :fn (:op %)) (vals decls)))))
200     :mutable #{}})

202  (defn- check-fn-decl
203    [decl decls imports]
204    (expr-walker/walk-expr (:body decl)
205                           (fn-ctx decl decls)
206                           (handlers decls imports)))

208  (defn check-module
209    [module]
210    (let [decls (decl-map module)
211          imports (imported-java-classes module)]
212      (doseq [decl (:decls module)]
213        (when (= :fn (:op decl))
214          (check-fn-decl decl decls imports)))
215      module))
