055    ([type-expr]
056     (lower-type type-expr nil))
057    ([type-expr ctx]
058     (or (get primitive-types type-expr)
059         (when (= 'String type-expr)
060           "java/lang/String")
061         (when (java-type? type-expr)
062           (java-internal-name type-expr))

150  (defn join-branch-types
151    [current next-branch data]
152    (cond
153      (bottom-types current) next-branch
154      (bottom-types next-branch) current
155      (= current next-branch) current
156      :else (fail! "Lowered branch types must agree."
157                   (assoc data
158                          :expected current
159                          :actual next-branch))))

209    [expr ctx]

245    [expr ctx]
246    (let [{:keys [body-ctx loop-types]}
247          (reduce (fn [{:keys [body-ctx loop-types]} binding]
248                    (let [binding-type (infer-type (:expr binding) body-ctx)]

253                  {:body-ctx ctx

258  (defn- infer-recur
259    [expr ctx]
260    (let [loop-types (:loop-types ctx)]
261      (when-not loop-types
262        (fail! "Recur used outside lowered loop."
263               {:expr expr}))
264      (when-not (= (count loop-types) (count (:args expr)))
265        (fail! "Lowered recur arity mismatch."
266               {:expected (count loop-types)
267                :actual (count (:args expr))}))
268      (doseq [[expected-type arg] (map vector loop-types (:args expr))]
269        (let [arg-type (infer-type arg ctx)]
270          (when-not (= expected-type arg-type)
271            (fail! "Lowered recur argument type mismatch."
272                   {:expected expected-type
273                    :actual arg-type
274                    :arg arg}))))
275      recur-type))

278    [expr ctx]
279    (let [body-type (infer-type (:body expr) ctx)
280          catch-types (mapv (fn [catch]
281                              (infer-type (:body catch)
282                                          (bind-local ctx (:name catch) (:type catch))))
283                            (:catches expr))]
284      (when-let [finally-expr (:finally expr)]
285        (infer-type finally-expr ctx))
286      (reduce (fn [current catch-type]
287                (join-branch-types current
288                                   catch-type
289                                   {:expr expr}))
290              body-type
291              catch-types)))
