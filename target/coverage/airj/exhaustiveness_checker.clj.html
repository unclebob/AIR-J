023    [decls]

037  (defn- enum-pattern?
038    [pattern target-type decls]
039    (patterns/enum-pattern? pattern target-type decls))

041  (defn- exhaustive-pattern?
042    [pattern target-type decls]
043    (or (= :wildcard-pattern (:op pattern))
044        (and (= :binder-pattern (:op pattern))
045             (not (enum-pattern? pattern target-type decls)))))
046  
047  (defn- bool-pattern-cover
048    [pattern]
049    (when (= :literal-pattern (:op pattern))
050      (:literal pattern)))
051  
052  (defn- enum-pattern-cover
053    [pattern target-type decls]
054    (when (enum-pattern? pattern target-type decls)
055      (:name pattern)))
056  
057  (defn- union-pattern-cover
058    [pattern]
059    (when (= :union-pattern (:op pattern))
060      (:name pattern)))
061  
062  (defn- pattern-cover
063    [pattern target-type decls]
064    (let [decl (patterns/decl-for-type decls target-type)]
065      (cond
066        (exhaustive-pattern? pattern target-type decls) :all
067        (= 'Bool target-type) (bool-pattern-cover pattern)
068        (= :enum (:op decl)) (enum-pattern-cover pattern target-type decls)
069        (= :union (:op decl)) (union-pattern-cover pattern)
070        :else nil)))
071  
072  (defn- bool-exhaustive?
073    [covers]
074    (and (contains? covers true)
075         (contains? covers false)))
076  
077  (defn- enum-exhaustive?
078    [decl covers]
079    (every? covers (:variants decl)))
080  
081  (defn- union-exhaustive?
082    [decl covers]
083    (every? covers (map :name (:variants decl))))
084  
085  (defn- supported-exhaustive-type?
086    [target-type decl]
087    (or (= 'Bool target-type)
088        (= :enum (:op decl))
089        (= :union (:op decl))))
090  
091  (defn- exhaustive-covers?
092    [target-type decl covers]
093    (or (contains? covers :all)
094        (and (= 'Bool target-type) (bool-exhaustive? covers))
095        (and (= :enum (:op decl)) (enum-exhaustive? decl covers))
096        (and (= :union (:op decl)) (union-exhaustive? decl covers))))

098  (defn- check-match
099    [expr ctx decls walk]
100    (let [target-type (type-checker/infer-expr-type (:target expr) ctx decls)
101          covers (set (keep #(pattern-cover (:pattern %) target-type decls)
102                            (:cases expr)))
103          decl (patterns/decl-for-type decls target-type)]
104      (when (and (supported-exhaustive-type? target-type decl)
105                 (not (exhaustive-covers? target-type decl covers)))
106        (fail! "Non-exhaustive match."
107               {:expr expr
108                :type target-type}))
109      (doseq [case (:cases expr)]
110        (walk (:body case)
111              (bind-pattern ctx (:pattern case) target-type decls)))))
112  
113  (defn- bind-literal-pattern
114    [ctx _pattern _target-type]
115    ctx)
116  
117  (defn- bind-binder-pattern
118    [ctx pattern target-type decls]
119    (if (enum-pattern? pattern target-type decls)
120      ctx
121      (assoc-in ctx [:locals (:name pattern)] target-type)))

123  (defn bind-pattern
124    [ctx pattern target-type decls]
125    (patterns/bind-pattern ctx
126                           pattern
127                           target-type
128                           decls
129                           {:bind-binder bind-binder-pattern
130                            :bind-literal bind-literal-pattern
131                            :fail! fail!}))

133  (defn- extend-let-ctx
134    [ctx bindings decls]
135    (reduce (fn [acc binding]
136              (assoc-in acc
137                        [:locals (:name binding)]
138                        (type-checker/infer-expr-type (:expr binding) acc decls)))
139            ctx

142  (defn- expr-checkers
143    [decls]
144    {:call (fn [expr ctx walk]
145             (walk (:callee expr) ctx)
146             (doseq [arg (:args expr)]
147               (walk arg ctx)))
148     :construct (fn [expr ctx walk]
149                  (doseq [arg (:args expr)]
150                    (walk arg ctx)))
151     :variant (fn [expr ctx walk]
152                (doseq [arg (:args expr)]
153                  (walk arg ctx)))
154     :record-get (fn [expr ctx walk]
155                   (walk (:target expr) ctx))
156     :if (fn [expr ctx walk]
157           (walk (:test expr) ctx)
158           (walk (:then expr) ctx)
159           (walk (:else expr) ctx))
160     :match (fn [expr ctx walk] (check-match expr ctx decls walk))
161     :let (fn [expr ctx walk]
162            (doseq [binding (:bindings expr)]
163              (walk (:expr binding) ctx))
164            (walk (:body expr)
165                  (extend-let-ctx ctx (:bindings expr) decls)))
166     :seq (fn [expr ctx walk]
167            (reduce (fn [current-ctx part]
168                      (walk part current-ctx)
169                      (if (= :var (:op part))
170                        (assoc-in current-ctx
171                                  [:locals (:name part)]
172                                  (type-checker/infer-expr-type part current-ctx decls))
173                        current-ctx))
174                    ctx
175                    (:exprs expr))
176            nil)
177     :lambda (fn [expr ctx walk]
178               (let [lambda-ctx (reduce (fn [acc param]
179                                          (assoc-in acc [:locals (:name param)] (:type param)))
180                                        ctx
181                                        (:params expr))]
182                 (walk (:body expr) lambda-ctx)))
183     :try (fn [expr ctx walk]
184            (walk (:body expr) ctx)
185            (doseq [catch (:catches expr)]
186              (walk (:body catch)
187                    (assoc-in ctx [:locals (:name catch)] (:type catch))))
188            (when-let [finally-expr (:finally expr)]
189              (walk finally-expr ctx)))
190     :var (fn [expr ctx walk]
191            (walk (:init expr) ctx))
192     :set (fn [expr ctx walk]
193            (walk (:expr expr) ctx))
194     :loop (fn [expr ctx walk]
195             (let [loop-ctx (extend-let-ctx ctx (:bindings expr) decls)]
196               (doseq [binding (:bindings expr)]
197                 (walk (:expr binding) ctx))
198               (walk (:body expr) loop-ctx)))
199     :recur (fn [expr ctx walk]
200              (doseq [arg (:args expr)]
201                (walk arg ctx)))
202     :raise (fn [expr ctx walk]
203              (walk (:expr expr) ctx))
204     :java-new (fn [expr ctx walk]
205                 (doseq [arg (:args expr)]
206                   (walk arg ctx)))
207     :java-call (fn [expr ctx walk]
208                  (walk (:target expr) ctx)
209                  (doseq [arg (:args expr)]
210                    (walk arg ctx)))
211     :java-static-call (fn [expr ctx walk]
212                         (doseq [arg (:args expr)]
213                           (walk arg ctx)))
214     :java-get-field (fn [expr ctx walk]
215                       (walk (:target expr) ctx))
216     :java-set-field (fn [expr ctx walk]
217                       (walk (:target expr) ctx)
218                       (walk (:expr expr) ctx))})

233    [module]
