001  ;; mutation-tested: 2026-03-11
002  (ns airj.java-members
003    (:require [airj.java-types :as java-types])
004    (:import (java.lang.reflect Constructor Field Method Modifier)))
005  
006  (defn- signature-types
007    [signature]
008    {:params (mapv java-types/resolve-type (:params signature))
009     :return-type (java-types/resolve-type (:return-type signature))})
010  
011  (defn- load-class
012    [class-name]
013    (try
014      (java-types/load-class class-name)
015      (catch ClassNotFoundException _
016        (throw (ex-info "Unknown Java class."
017                        {:class-name class-name})))))
018  
019  (defn- method-meta
020    [class-name ^Method method member-id]
021    {:kind :method
022     :class-name class-name
023     :member-id member-id
024     :static? (Modifier/isStatic (.getModifiers method))
025     :parameter-types (vec (.getParameterTypes method))
026     :return-type (.getReturnType method)
027     :declaring-class (.getDeclaringClass method)})
028  
029  (defn- field-meta
030    [class-name ^Field field field-name]
031    {:kind :field
032     :class-name class-name
033     :field-name field-name
034     :static? (Modifier/isStatic (.getModifiers field))
035     :field-type (.getType field)
036     :declaring-class (.getDeclaringClass field)})
037  
038  (defn- constructor-meta
039    [class-name ^Constructor ctor]
040    {:kind :constructor
041     :class-name class-name
042     :parameter-types (vec (.getParameterTypes ctor))
043     :declaring-class (.getDeclaringClass ctor)})
044  
045  (defn resolve-constructor
046    [class-name arg-types]
047    (let [klass (load-class class-name)]
048      (some (fn [^Constructor ctor]
049              (when (and (= (count arg-types) (alength (.getParameterTypes ctor)))
050                         (every? true? (map = arg-types (.getParameterTypes ctor))))
051                (constructor-meta class-name ctor)))
052            (.getConstructors klass))))
053  
054  (defn resolve-static-method
055    [class-name member-id signature]
056    (let [klass (load-class class-name)
057          {:keys [params return-type]} (signature-types signature)]
058      (some (fn [^Method method]
059              (when (and (= (name member-id) (.getName method))
060                         (Modifier/isStatic (.getModifiers method))
061                         (= (count params) (alength (.getParameterTypes method)))
062                         (every? true? (map = params (.getParameterTypes method)))
063                         (= return-type (.getReturnType method)))
064                (method-meta class-name method member-id)))
065            (.getMethods klass))))
066  
067  (defn resolve-instance-method
068    [receiver-type member-id signature]
069    (let [klass (java-types/resolve-type receiver-type)
070          class-name (second receiver-type)
071          {:keys [params return-type]} (signature-types signature)]
072      (some (fn [^Method method]
073              (when (and (= (name member-id) (.getName method))
074                         (not (Modifier/isStatic (.getModifiers method)))
075                         (= (count params) (alength (.getParameterTypes method)))
076                         (every? true? (map = params (.getParameterTypes method)))
077                         (= return-type (.getReturnType method)))
078                (method-meta class-name method member-id)))
079            (.getMethods klass))))
080  
081  (defn resolve-instance-field
082    [receiver-type field-name field-type]
083    (let [klass (java-types/resolve-type receiver-type)
084          class-name (second receiver-type)
085          resolved-type (java-types/resolve-type field-type)]
086      (some (fn [^Field field]
087              (when (and (= (name field-name) (.getName field))
088                         (not (Modifier/isStatic (.getModifiers field)))
089                         (= resolved-type (.getType field)))
090                (field-meta class-name field field-name)))
091            (.getFields klass))))
