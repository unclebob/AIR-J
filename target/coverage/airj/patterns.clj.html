001  ;; mutation-tested: 2026-03-11
002  (ns airj.patterns)
003  
004  (declare bind-pattern)
005  
006  (defn decl-for-type
007    [decls type-expr]
008    (get decls type-expr))
009  
010  (defn field-type
011    [decl field-name]
012    (some (fn [field]
013            (when (= field-name (:name field))
014              (:type field)))
015          (:fields decl)))
016  
017  (defn union-variant
018    [decl variant-name]
019    (some #(when (= (:name %) variant-name) %) (:variants decl)))
020  
021  (defn enum-variant?
022    [decl variant-name]
023    (boolean (some #{variant-name} (:variants decl))))
024  
025  (defn enum-pattern?
026    [pattern target-type decls]
027    (let [decl (decl-for-type decls target-type)]
028      (and (= :binder-pattern (:op pattern))
029           (= :enum (:op decl))
030           (enum-variant? decl (:name pattern)))))
031  
032  (defn- options
033    [bind-binder bind-literal fail!]
034    {:bind-binder bind-binder
035     :bind-literal bind-literal
036     :fail! fail!})
037  
038  (defn- bind-union-pattern
039    [ctx pattern target-type decls opts]
040    (let [decl (decl-for-type decls target-type)]
041      (when-not (= :union (:op decl))
042        ((:fail! opts) "Expected union type."
043                      {:pattern pattern
044                       :type target-type}))
045      (let [variant (union-variant decl (:name pattern))
046            expected-count (count (:fields variant))
047            actual-count (count (:args pattern))]
048        (when-not variant
049          ((:fail! opts) "Unknown variant."
050                        {:pattern pattern
051                         :type target-type}))
052        (when-not (= expected-count actual-count)
053          ((:fail! opts) "Arity mismatch."
054                        {:pattern pattern
055                         :type target-type
056                         :expected expected-count
057                         :actual actual-count}))
058        (reduce (fn [acc [field nested-pattern]]
059                  (bind-pattern acc
060                                nested-pattern
061                                (:type field)
062                                decls
063                                opts))
064                ctx
065                (map vector (:fields variant) (:args pattern))))))
066  
067  (defn- bind-record-field
068    [ctx record-decl field-pattern decls opts target-type]
069    (let [matched-type (field-type record-decl (:name field-pattern))]
070      (when-not matched-type
071        ((:fail! opts) "Unknown field."
072                      {:type target-type
073                       :field (:name field-pattern)}))
074      (bind-pattern ctx
075                    (:pattern field-pattern)
076                    matched-type
077                    decls
078                    opts)))
079  
080  (defn- bind-record-pattern
081    [ctx pattern target-type decls opts]
082    (when-not (= (:type pattern) target-type)
083      ((:fail! opts) "Type mismatch."
084                    {:pattern pattern
085                     :expected target-type
086                     :actual (:type pattern)}))
087    (let [record-decl (decl-for-type decls target-type)]
088      (when-not (= :data (:op record-decl))
089        ((:fail! opts) "Expected record type."
090                      {:pattern pattern
091                       :type target-type}))
092      (reduce (fn [acc field-pattern]
093                (bind-record-field acc record-decl field-pattern decls opts target-type))
094              ctx
095              (:fields pattern))))
096  
097  (defn bind-pattern
098    [ctx pattern target-type decls {:keys [bind-binder bind-literal fail!]}]
099    (let [opts (options bind-binder bind-literal fail!)]
100      (case (:op pattern)
101        :wildcard-pattern ctx
102        :literal-pattern (bind-literal ctx pattern target-type)
103        :binder-pattern (bind-binder ctx pattern target-type decls)
104        :union-pattern (bind-union-pattern ctx pattern target-type decls opts)
105        :record-pattern (bind-record-pattern ctx pattern target-type decls opts)
106        (fail! "Unsupported pattern."
107               {:pattern pattern}))))
