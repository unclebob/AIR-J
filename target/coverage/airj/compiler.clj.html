103    [bundle output-dir]
104    (doseq [[internal-name bytecode] bundle]
105      (let [file (class-file output-dir internal-name)]
106        (io/make-parents file)
107        (with-open [out (io/output-stream file)]
108          (.write out bytecode))))
109    (into {}

111                 [internal-name (.getPath (class-file output-dir internal-name))]))
112          bundle))
