001  (ns airj.jvm-emit-entry
002    (:import (clojure.asm ClassWriter Opcodes)))
003  
004  (defn exported-main
005    [module]
006    (when (some #{'main} (:exports module))
007      (some (fn [method]
008              (when (and (= 'main (:name method))
009                         (or (empty? (:params method))
010                             (= [{:name 'args
011                                  :jvm-type "[Ljava/lang/String;"}]
012                                (:params method))))
013                method))
014            (:methods module))))
015  
016  (defn emit-main-wrapper
017    [^ClassWriter cw module {:keys [method-name method-descriptor pop-opcode]}]
018    (when-let [airj-main (exported-main module)]
019      (let [mv (.visitMethod cw
020                             (+ Opcodes/ACC_PUBLIC Opcodes/ACC_STATIC)
021                             "main"
022                             "([Ljava/lang/String;)V"
023                             nil
024                             nil)]
025        (.visitCode mv)
026        (when (= [{:name 'args
027                   :jvm-type "[Ljava/lang/String;"}]
028                 (:params airj-main))
029          (.visitVarInsn mv Opcodes/ALOAD 0))
030        (.visitMethodInsn mv
031                          Opcodes/INVOKESTATIC
032                          (:internal-name module)
033                          (method-name (:name airj-main))
034                          (method-descriptor (:params airj-main) (:return-type airj-main))
035                          false)
036        (case (:return-type airj-main)
037          :int (.visitMethodInsn mv
038                                 Opcodes/INVOKESTATIC
039                                 "java/lang/System"
040                                 "exit"
041                                 "(I)V"
042                                 false)
043          (when-let [opcode (pop-opcode (:return-type airj-main))]
044            (.visitInsn mv opcode)))
045        (.visitInsn mv Opcodes/RETURN)
046        (.visitMaxs mv 0 0)
047        (.visitEnd mv))))
