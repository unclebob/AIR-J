001  (ns airj.jvm-emit-cells
002    (:import (clojure.asm MethodVisitor Opcodes)))
003  
004  (def ^:private primitive-newarray-opcodes
005    {:int Opcodes/T_INT
006     :boolean Opcodes/T_BOOLEAN})
007  
008  (def ^:private array-load-opcodes
009    {:int Opcodes/IALOAD
010     :boolean Opcodes/BALOAD})
011  
012  (def ^:private array-store-opcodes
013    {:int Opcodes/IASTORE
014     :boolean Opcodes/BASTORE})
015  
016  (defn- array-load-opcode
017    [value-jvm-type]
018    (get array-load-opcodes value-jvm-type Opcodes/AALOAD))
019  
020  (defn- array-store-opcode
021    [value-jvm-type]
022    (get array-store-opcodes value-jvm-type Opcodes/AASTORE))
023  
024  (defn- newarray-opcode
025    [value-jvm-type]
026    (get primitive-newarray-opcodes value-jvm-type))
027  
028  (defn emit-cell-new
029    [^MethodVisitor mv expr env {:keys [emit-expr]}]
030    (.visitInsn mv Opcodes/ICONST_1)
031    (if-let [opcode (newarray-opcode (:value-jvm-type expr))]
032      (.visitIntInsn mv Opcodes/NEWARRAY opcode)
033      (.visitTypeInsn mv
034                      Opcodes/ANEWARRAY
035                      (if (and (string? (:value-jvm-type expr))
036                               (.startsWith ^String (:value-jvm-type expr) "["))
037                        (:value-jvm-type expr)
038                        (:value-jvm-type expr))))
039    (.visitInsn mv Opcodes/DUP)
040    (.visitInsn mv Opcodes/ICONST_0)
041    (emit-expr mv (:init expr) env)
042    (.visitInsn mv (array-store-opcode (:value-jvm-type expr))))
043  
044  (defn emit-cell-get
045    [^MethodVisitor mv expr env]
046    (let [{:keys [slot]} (get-in env [:slots (:name expr)])]
047      (.visitVarInsn mv Opcodes/ALOAD slot)
048      (.visitInsn mv Opcodes/ICONST_0)
049      (.visitInsn mv (array-load-opcode (:jvm-type expr)))))
050  
051  (defn emit-cell-set
052    [^MethodVisitor mv expr env {:keys [emit-expr]}]
053    (let [{:keys [slot]} (get-in env [:slots (:name expr)])]
054      (.visitVarInsn mv Opcodes/ALOAD slot)
055      (.visitInsn mv Opcodes/ICONST_0)
056      (emit-expr mv (:expr expr) env)
057      (.visitInsn mv (array-store-opcode (:value-jvm-type expr)))))
