010  (defn- fail!
011    [message data]
012    (throw (ex-info message data)))

014  (defn- decl-map
015    [module]
016    (into {} (map (juxt :name identity) (:decls module))))
017  
018  (defn- ensure-subset
019    [actual expected data]
020    (let [actual-set (set actual)
021          expected-set (set expected)
022          undeclared (set/difference actual-set expected-set)]
023      (when (seq undeclared)
024        (fail! "Undeclared effects."
025               (assoc data :undeclared (sort undeclared))))))

027  (defn- make-ctx
028    [module decls]
029    {:call-effects
030     (merge
031      (into {}
032            (map (fn [decl]
033                   [(:name decl) (set (:effects decl))])
034                 (filter #(= :fn (:op %)) (vals decls))))
035      (into {}
036            (keep (fn [[symbol {:keys [decl]}]]
037                    (when (= :fn (:op decl))
038                      [symbol (set (:effects decl))])))
039            (imported-interfaces/imported-decls module)))})

041  (defn- bind-call-effects
042    [ctx name effects]
043    (assoc-in ctx [:call-effects name] (set effects)))

045  (declare callable-effects)

047  (defn- local-callable-effects
048    [expr ctx decls]
049    (cond
050      (contains? (:call-effects ctx) (:name expr))
051      (get-in ctx [:call-effects (:name expr)])
052  
053      (= :fn (:op (get decls (:name expr))))
054      (set (:effects (get decls (:name expr))))
055  
056      :else nil))
057  
058  (defn- if-callable-effects
059    [expr ctx decls]
060    (let [then-effects (callable-effects (:then expr) ctx decls)
061          else-effects (callable-effects (:else expr) ctx decls)]
062      (when (and then-effects else-effects (= then-effects else-effects))
063        then-effects)))
064  
065  (defn- callable-effects
066    [expr ctx decls]
067    (cond
068      (= :local (:op expr))
069      (local-callable-effects expr ctx decls)
070  
071      (= :lambda (:op expr))
072      (set (:effects expr))
073  
074      (= :if (:op expr))
075      (if-callable-effects expr ctx decls)
076  
077      (= :seq (:op expr))
078      (callable-effects (last (:exprs expr)) ctx decls)
079  
080      :else nil))
081  
082  (defn- binding-call-effects
083    [binding ctx decls]
084    (callable-effects (:expr binding) ctx decls))

086  (defn- effect-union
087    [exprs ctx walk]
088    (apply set/union #{} (map #(walk % ctx) exprs)))
089  
090  (defn- call-effects
091    [expr ctx decls]
092    (or (callable-effects (:callee expr) ctx decls)
093        (fail! "Unsupported call effect inference."
094               {:expr expr})))
095  
096  (defn- lambda-effects
097    [expr ctx walk]
098    (let [body-effects (walk (:body expr) ctx)]
099      (ensure-subset body-effects
100                     (:effects expr)
101                     {:lambda expr})
102      #{}))
103  
104  (defn- try-effects
105    [expr ctx walk]
106    (set/union (walk (:body expr) ctx)
107               (effect-union (map :body (:catches expr)) ctx walk)
108               (if-let [finally-expr (:finally expr)]
109                 (walk finally-expr ctx)
110                 #{})))
111  
112  (defn- let-effects
113    [expr ctx decls walk]
114    (let [{:keys [effects ctx]}
115          (reduce (fn [acc binding]
116                    (let [binding-effects (walk (:expr binding) (:ctx acc))
117                          updated-ctx (if-let [bound-effects (binding-call-effects binding (:ctx acc) decls)]
118                                        (bind-call-effects (:ctx acc) (:name binding) bound-effects)
119                                        (:ctx acc))]
120                      {:effects (set/union (:effects acc) binding-effects)
121                       :ctx updated-ctx}))
122                  {:effects #{}
123                   :ctx ctx}
124                  (:bindings expr))]
125      (set/union effects
126                 (walk (:body expr) ctx))))

128  (defn- expr-effect-handlers
129    [decls]
130    {:call (fn [expr ctx walk]
131             (set/union (walk (:callee expr) ctx)
132                        (effect-union (:args expr) ctx walk)
133                        (call-effects expr ctx decls)))
134     :construct (fn [expr ctx walk] (effect-union (:args expr) ctx walk))
135     :variant (fn [expr ctx walk] (effect-union (:args expr) ctx walk))
136     :record-get (fn [expr ctx walk] (walk (:target expr) ctx))
137     :if (fn [expr ctx walk]
138           (set/union (walk (:test expr) ctx)
139                      (walk (:then expr) ctx)
140                      (walk (:else expr) ctx)))
141     :match (fn [expr ctx walk]
142              (set/union (walk (:target expr) ctx)
143                         (effect-union (map :body (:cases expr)) ctx walk)))
144     :let (fn [expr ctx walk] (let-effects expr ctx decls walk))
145     :seq (fn [expr ctx walk] (effect-union (:exprs expr) ctx walk))
146     :lambda lambda-effects
147     :try try-effects
148     :var (fn [expr ctx walk]
149            (set/union #{'State.Write}
150                       (walk (:init expr) ctx)))
151     :set (fn [expr ctx walk]
152            (set/union #{'State.Write}
153                       (walk (:expr expr) ctx)))
154     :loop (fn [expr ctx walk]
155             (set/union (effect-union (map :expr (:bindings expr)) ctx walk)
156                        (walk (:body expr) ctx)))
157     :recur (fn [expr ctx walk] (effect-union (:args expr) ctx walk))
158     :raise (fn [expr ctx walk]
159              (set/union #{'Foreign.Throw}
160                         (walk (:expr expr) ctx)))
161     :java-new (fn [expr ctx walk]
162                 (let [arg-effects (effect-union (:args expr) ctx walk)
163                       arg-types (mapv #(java-types/resolve-type (type-checker/infer-expr-type % ctx decls))
164                                       (:args expr))
165                       member (java-members/resolve-constructor (:class-name expr) arg-types)]
166                   (set/union (java-effects/member-effects member)
167                              arg-effects)))
168     :java-call (fn [expr ctx walk]
169                  (let [target-effects (walk (:target expr) ctx)
170                        arg-effects (effect-union (:args expr) ctx walk)
171                        receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
172                        member (java-members/resolve-instance-method receiver-type
173                                                                     (:member-id expr)
174                                                                     (:signature expr))]
175                    (set/union (java-effects/member-effects member)
176                               target-effects
177                               arg-effects)))
178     :java-static-call (fn [expr ctx walk]
179                         (let [arg-effects (effect-union (:args expr) ctx walk)
180                               member (java-members/resolve-static-method (:class-name expr)
181                                                                          (:member-id expr)
182                                                                          (:signature expr))]
183                           (set/union (java-effects/member-effects member)
184                                      arg-effects)))
185     :java-get-field (fn [expr ctx walk]
186                       (let [target-effects (walk (:target expr) ctx)
187                             receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
188                             member (java-members/resolve-instance-field receiver-type
189                                                                        (:field-name expr)
190                                                                        (:field-type expr))]
191                         (set/union (java-effects/member-effects member)
192                                    target-effects)))
193     :java-set-field (fn [expr ctx walk]
194                       (let [target-effects (walk (:target expr) ctx)
195                             value-effects (walk (:expr expr) ctx)
196                             receiver-type (type-checker/infer-expr-type (:target expr) ctx decls)
197                             member (java-members/resolve-instance-field receiver-type
198                                                                        (:field-name expr)
199                                                                        (:field-type expr))]
200                         (set/union (java-effects/interop-effects expr member)
201                                    target-effects
202                                    value-effects)))})

204  (defn expr-effects*
205    [expr ctx decls]
206    (or (expr-walker/walk-expr expr ctx (expr-effect-handlers decls))
207        #{}))

209  (defn expr-effects
210    [expr decls]
211    (expr-effects* expr {:call-effects {}}
212                   decls))

214  (defn- check-contracts
215    [decl ctx decls]
216    (doseq [contract (concat (:requires decl) (:ensures decl))]
217      (when (seq (expr-effects* contract ctx decls))
218        (fail! "Contracts must be pure."
219               {:fn (:name decl)
220                :contract contract}))))
221  
222  (defn- check-fn-decl
223    [decl ctx decls]
224    (check-contracts decl ctx decls)
225    (ensure-subset (expr-effects* (:body decl) ctx decls)
226                   (:effects decl)
227                   {:fn (:name decl)}))

229  (defn check-module
230    [module]
231    (let [decls (decl-map module)
232          ctx (make-ctx module decls)]
233      (doseq [decl (:decls module)]
234        (when (= :fn (:op decl))
235          (check-fn-decl decl ctx decls)))
236      module))
