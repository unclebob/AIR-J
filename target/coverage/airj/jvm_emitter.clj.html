331  (defn- emit-match
332    [^MethodVisitor mv expr env]
333    (let [end-label (Label.)
334          case-labels (repeatedly (count (:cases expr)) #(Label.))
335          next-labels (concat (rest case-labels) [nil])]
336      (doseq [[[case current-label] next-label] (map vector (map vector (:cases expr) case-labels) next-labels)]
337        (.visitLabel mv current-label)
338        (emit-expr mv (:test case) env)
339        (when next-label
340          (.visitJumpInsn mv Opcodes/IFEQ next-label))
341        (when-not next-label
342          (.visitInsn mv Opcodes/POP))
343        (let [body-env (bind-local-slots env (:bindings case))]
344          (doseq [binding (:bindings case)]
345            (emit-binding mv binding body-env))
346          (emit-expr mv (:body case) body-env))
347        (.visitJumpInsn mv Opcodes/GOTO end-label))
348      (.visitLabel mv end-label)))

366  (defn- emit-recur
367    [^MethodVisitor mv expr env]
368    (let [{:keys [start-label bindings]} (recur-frame env)
369          {:keys [slots]} (temp-slots env (:args expr))]
370      (doseq [[arg {:keys [slot jvm-type]}] (map vector (:args expr) slots)]
371        (emit-expr mv arg env)
372        (.visitVarInsn mv (store-opcode jvm-type) slot))
373      (doseq [[{:keys [slot jvm-type]} {:keys [binding-slot]}]

379                        bindings))]
380        (.visitVarInsn mv (load-opcode jvm-type) slot)

408    [^MethodVisitor mv expr env]
409    (let [start-label (Label.)
410          loop-env (bind-loop-slots env (:bindings expr))
411          binding-state (mapv (fn [binding]
412                                (merge binding
413                                       (get-in loop-env [:slots (:name binding)])))
414                              (:bindings expr))

419                (.visitVarInsn mv
420                               (store-opcode (:jvm-type binding))
421                               (:slot binding))
422                (assoc-in current-env
423                          [:slots (:name binding)]
424                          {:slot (:slot binding)
425                           :jvm-type (:jvm-type binding)}))
426              env
427              binding-state)
428      (.visitLabel mv start-label)
429      (emit-expr mv (:body expr) body-env)))
