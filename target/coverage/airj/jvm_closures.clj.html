218  (defn lower-lambda
219    [expr ctx {:keys [fail! lower-type lower-expr lower-param bind-local]}]
220    (let [param-types (mapv #(lower-type (:type %) ctx) (:params expr))
221          return-type (lower-type (:return-type expr) ctx)
222          interface-name (register-closure-interface! ctx param-types return-type fail!)
223          class-name (next-closure-class-name! ctx)
224          param-names (into #{} (map :name) (:params expr))
225          capture-names (filterv #(and (contains? (:locals ctx) %)
226                                       (not (contains? param-names %)))
227                                 (free-locals (:body expr) param-names))
228          mutable-locals (:mutable-locals ctx)
229          captures (mapv (fn [name]
230                           (let [type-expr (get-in ctx [:locals name])
231                                 mutable? (contains? mutable-locals name)

233                             {:name name
234                              :type-expr type-expr
235                              :mutable? mutable?
236                              :jvm-type (if mutable?
237                                          (jvm-cells/cell-jvm-type value-jvm-type)
238                                          value-jvm-type)}))
239                         capture-names)
240          lambda-ctx (reduce (fn [current capture]
241                               (cond-> (bind-local current
242                                                   (:name capture)
243                                                   (:type-expr capture))
244                                 (:mutable? capture)
245                                 (assoc-in [:mutable-locals (:name capture)]
246                                           (:type-expr capture))))
247                             (assoc ctx
248                                    :locals {}
249                                    :lambdas {}
250                                    :mutable-locals {})
251                             captures)
252          lambda-ctx (reduce (fn [current param]
253                               (bind-local current (:name param) (:type param)))
254                             lambda-ctx
255                             (:params expr))]
256      (swap! (:closures ctx) conj
257             {:class-name class-name
258              :interface-name interface-name
259              :method-name "apply"
260              :captures (mapv #(select-keys % [:name :jvm-type]) captures)
261              :params (mapv #(lower-param % ctx) (:params expr))
262              :return-type return-type
