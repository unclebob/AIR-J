001  (ns airj.jvm-emit-try
002    (:import (clojure.asm Label MethodVisitor Opcodes)))
003  
004  (defn- try-layout
005    [expr env temp-slot]
006    (let [result-needed? (not= :void (:jvm-type expr))
007          result-slot (when result-needed?
008                        (temp-slot env (:jvm-type expr)))
009          env-after-result (or (:env result-slot) env)
010          throwable-slot (when (:finally expr)
011                           (temp-slot env-after-result "java/lang/Throwable"))
012          catch-count (count (:catches expr))]
013      {:start-label (Label.)
014       :end-label (Label.)
015       :done-label (Label.)
016       :finally-label (when (:finally expr) (Label.))
017       :catch-labels (vec (repeatedly catch-count #(Label.)))
018       :catch-ranges (vec (repeatedly catch-count
019                                      #(when (:finally expr)
020                                         [(Label.) (Label.)])))
021       :result-slot result-slot
022       :throwable-slot throwable-slot
023       :base-env (or (:env throwable-slot) env-after-result env)}))
024  
025  (defn- emit-try-registrations
026    [^MethodVisitor mv expr {:keys [start-label end-label finally-label catch-labels catch-ranges]}]
027    (doseq [[handler catch-clause] (map vector catch-labels (:catches expr))]
028      (.visitTryCatchBlock mv start-label end-label handler (:type catch-clause)))
029    (when finally-label
030      (.visitTryCatchBlock mv start-label end-label finally-label nil)
031      (doseq [[catch-start catch-end] catch-ranges]
032        (.visitTryCatchBlock mv catch-start catch-end finally-label nil))))
033  
034  (defn- emit-result-store
035    [^MethodVisitor mv result-slot jvm-type store-opcode]
036    (when result-slot
037      (.visitVarInsn mv (store-opcode jvm-type) (:slot result-slot))))
038  
039  (defn- emit-result-load
040    [^MethodVisitor mv result-slot jvm-type load-opcode]
041    (when result-slot
042      (.visitVarInsn mv (load-opcode jvm-type) (:slot result-slot))))
043  
044  (defn- catch-env
045    [base-env catch-clause throwable-slot]
046    (let [slot-index (or (:slot throwable-slot)
047                         (:next-slot base-env))
048          env-with-slot (if throwable-slot
049                          base-env
050                          (update base-env :next-slot inc))]
051      {:slot-index slot-index
052       :env (assoc-in env-with-slot
053                      [:slots (:name catch-clause)]
054                      {:slot slot-index
055                       :jvm-type (:type catch-clause)})}))
056  
057  (defn- emit-try-success
058    [^MethodVisitor mv expr layout tools]
059    (.visitLabel mv (:start-label layout))
060    ((:emit-expr tools) mv (:body expr) (:base-env layout))
061    (emit-result-store mv (:result-slot layout) (:jvm-type expr) (:store-opcode tools))
062    (.visitLabel mv (:end-label layout))
063    (when-let [finally-expr (:finally expr)]
064      ((:emit-discarded tools) mv finally-expr (:base-env layout)))
065    (emit-result-load mv (:result-slot layout) (:jvm-type expr) (:load-opcode tools))
066    (.visitJumpInsn mv Opcodes/GOTO (:done-label layout)))
067  
068  (defn- emit-try-catch
069    [^MethodVisitor mv expr layout catch-clause handler-label catch-range tools]
070    (.visitLabel mv handler-label)
071    (let [{:keys [slot-index env]} (catch-env (:base-env layout)
072                                              catch-clause
073                                              (:throwable-slot layout))]
074      (.visitVarInsn mv Opcodes/ASTORE slot-index)
075      (when-let [[catch-start _] catch-range]
076        (.visitLabel mv catch-start))
077      ((:emit-expr tools) mv (:body catch-clause) env)
078      (emit-result-store mv (:result-slot layout) (:jvm-type expr) (:store-opcode tools))
079      (when-let [[_ catch-end] catch-range]
080        (.visitLabel mv catch-end))
081      (when-let [finally-expr (:finally expr)]
082        ((:emit-discarded tools) mv finally-expr env))
083      (emit-result-load mv (:result-slot layout) (:jvm-type expr) (:load-opcode tools))
084      (.visitJumpInsn mv Opcodes/GOTO (:done-label layout))))
085  
086  (defn- emit-try-finally-handler
087    [^MethodVisitor mv expr layout tools]
088    (when-let [finally-expr (:finally expr)]
089      (.visitLabel mv (:finally-label layout))
090      (.visitVarInsn mv Opcodes/ASTORE (:slot (:throwable-slot layout)))
091      ((:emit-discarded tools) mv finally-expr (:base-env layout))
092      (.visitVarInsn mv Opcodes/ALOAD (:slot (:throwable-slot layout)))
093      (.visitInsn mv Opcodes/ATHROW)))
094  
095  (defn emit-try
096    [^MethodVisitor mv expr env tools]
097    (let [layout (try-layout expr env (:temp-slot tools))]
098      (emit-try-registrations mv expr layout)
099      (emit-try-success mv expr layout tools)
100      (doseq [[catch-clause handler-label catch-range]
101              (map vector
102                   (:catches expr)
103                   (:catch-labels layout)
104                   (:catch-ranges layout))]
105        (emit-try-catch mv expr layout catch-clause handler-label catch-range tools))
106      (emit-try-finally-handler mv expr layout tools)
107      (.visitLabel mv (:done-label layout))))
