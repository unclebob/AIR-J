098    [expected actual]

104    [expected actual context]

152    [expr ctx]

159  (defn- infer-record-get
160    [expr ctx decls]
161    (let [target-type (infer-expr-type (:target expr) ctx decls)
162          target-decl (get decls target-type)
163          result-type (and target-decl (patterns/field-type target-decl (:field expr)))]
164      (when-not target-decl
165        (fail! "Expected record type."
166               {:type target-type}))
167      (result (or result-type
168                  (fail! "Unknown field."
169                         {:type target-type
170                          :field (:field expr)}))
171              ctx)))

173  (defn- infer-construct
174    [expr ctx decls]
175    (let [decl (get decls (:type expr))]
176      (when-not decl
177        (fail! "Unknown constructed type."
178               {:type (:type expr)}))
179      (ensure-arity= (count (:fields decl))
180                     (count (:args expr))
181                     {:type (:type expr)})
182      (doseq [[field arg] (map vector (:fields decl) (:args expr))]
183        (ensure-type= (:type field)
184                      (infer-expr-type arg ctx decls)
185                      {:type (:type expr)
186                       :field (:name field)}))
187      (result (:type expr) ctx)))

189  (defn- infer-variant
190    [expr ctx decls]
191    (let [decl (get decls (:type expr))]
192      (when-not decl
193        (fail! "Unknown constructed type."
194               {:type (:type expr)}))
195      (when-not (= :union (:op decl))
196        (fail! "Expected union type."
197               {:type (:type expr)}))
198      (let [variant (patterns/union-variant decl (:name expr))]
199        (when-not variant
200          (fail! "Unknown variant."
201                 {:type (:type expr)
202                  :variant (:name expr)}))
203        (ensure-arity= (count (:fields variant))
204                       (count (:args expr))
205                       {:type (:type expr)
206                        :variant (:name expr)})
207        (doseq [[field arg] (map vector (:fields variant) (:args expr))]
208          (ensure-type= (:type field)
209                        (infer-expr-type arg ctx decls)
210                        {:type (:type expr)
211                         :variant (:name expr)
212                         :field (:name field)}))
213        (result (:type expr) ctx))))

215  (defn- infer-if
216    [expr ctx decls]
217    (let [test-type (infer-expr-type (:test expr) ctx decls)
218          then-type (infer-expr-type (:then expr) ctx decls)
219          else-type (infer-expr-type (:else expr) ctx decls)]
220      (when-not (= 'Bool test-type)
221        (fail! "Expected Bool."
222               {:actual test-type}))
223      (result (join-types then-type else-type {:op :if}) ctx)))

225  (defn- infer-match
226    [expr ctx decls]
227    (let [target-type (infer-expr-type (:target expr) ctx decls)
228          case-types (mapv (fn [case]
229                             (let [case-ctx (bind-pattern ctx
230                                                          (:pattern case)
231                                                          target-type
232                                                          decls)]
233                               (infer-expr-type (:body case) case-ctx decls)))
234                           (:cases expr))]
235      (result (reduce (fn [current case-type]
236                        (join-types current case-type {:expr expr}))
237                      (first case-types)
238                      (rest case-types))
239              ctx)))

241  (defn- infer-call
242    [expr ctx decls]
243    (let [callee-type (infer-expr-type (:callee expr) ctx decls)]
244      (when-not (= :fn-type (:op callee-type))
245        (fail! "Expected function type."
246               {:actual callee-type}))
247      (ensure-arity= (count (:params callee-type))
248                     (count (:args expr))
249                     {:callee (:callee expr)})
250      (doseq [[expected-type arg] (map vector (:params callee-type) (:args expr))]
251        (ensure-type= expected-type
252                      (infer-expr-type arg ctx decls)
253                      {:callee (:callee expr)
254                       :arg arg}))
255      (result (:return-type callee-type) ctx)))

259    (let [lambda-ctx (reduce (fn [acc param]

261                             ctx

264      (ensure-type= (:return-type expr)
265                    body-type
266                    {:lambda expr})
267      (result (fn-type (map :type (:params expr))
268                       (:return-type expr)
269                       (:effects expr))
270              ctx)))

312  (defn- infer-loop
313    [expr ctx decls]
314    (let [binding-results (reduce (fn [acc binding]
315                                    (let [binding-type (infer-expr-type (:expr binding)
316                                                                        (:ctx acc)
317                                                                        decls)]
318                                      {:ctx (bind-local (:ctx acc)
319                                                        (:name binding)
320                                                        binding-type)
321                                       :types (conj (:types acc) binding-type)}))
322                                  {:ctx ctx
323                                   :types []}
324                                  (:bindings expr))
325          loop-ctx (with-loop-types (:ctx binding-results) (:types binding-results))
326          body-type (infer-expr-type (:body expr) loop-ctx decls)]
327      (result body-type ctx)))

329  (defn- infer-recur
330    [expr ctx decls]
331    (let [loop-types (:loop-types ctx)]
332      (when-not loop-types
333        (fail! "Recur used outside loop."
334               {:expr expr}))
335      (ensure-arity= (count loop-types)
336                     (count (:args expr))
337                     {:expr expr})
338      (doseq [[expected-type arg] (map vector loop-types (:args expr))]
339        (ensure-type= expected-type
340                      (infer-expr-type arg (without-loop-types ctx) decls)
341                      {:expr expr
342                       :arg arg}))
343      (result (bottom-type :recur) ctx)))
344  
345  (defn- infer-try
346    [expr ctx decls]
347    (let [body-type (infer-expr-type (:body expr) ctx decls)
348          catch-types (mapv (fn [catch]
349                              (let [catch-ctx (bind-local ctx (:name catch) (:type catch))]
350                                (infer-expr-type (:body catch) catch-ctx decls)))
351                            (:catches expr))]
352      (doseq [catch-type catch-types]
353        (ensure-type= body-type catch-type {:expr expr}))
354      (when-let [finally-expr (:finally expr)]
355        (infer-expr-type finally-expr ctx decls))
356      (result (reduce (fn [current catch-type]
357                        (join-types current catch-type {:expr expr}))
358                      body-type
359                      catch-types)
360              ctx)))

443  (defn infer-expr
444    [expr ctx decls]
445    (if-let [type-expr (literal-type expr)]
446      (result type-expr ctx)
447      (if-let [inferer (get expr-inferers (:op expr))]
448        (inferer expr ctx decls)
449        (fail! "Unsupported type inference."
450               {:expr expr}))))

462  (defn- check-fn-decl
463    [decl module decls]
464    (let [ctx (reduce (fn [acc param]
465                        (bind-local acc (:name param) (:type param)))
466                      (make-ctx (module-fn-env module decls))
467                      (:params decl))
468          body-type (infer-expr-type (:body decl) ctx decls)]
469      (doseq [contract (concat (:requires decl) (:ensures decl))]
470        (check-bool-contract contract
471                             ctx
472                             decls
473                             {:fn (:name decl)
474                              :contract contract}))

476                    body-type
477                    {:fn (:name decl)})))

480    [module]
