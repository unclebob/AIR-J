001  (ns airj.resolver
002    (:require [airj.expr-walker :as expr-walker]
003              [airj.imported-interfaces :as imported-interfaces]
004              [clojure.set :as set]))

006  (declare pattern-binders)

008  (defn- fail!
009    [message data]
010    (throw (ex-info message data)))
011  
012  (defn- declared-symbols
013    [module]
014    (set (map :name (:decls module))))
015  
016  (defn- declared-function-symbols
017    [module]
018    (set (map :name (filter #(= :fn (:op %)) (:decls module)))))
019  
020  (defn- imported-function-symbols
021    [module]
022    (->> (imported-interfaces/imported-decls module)
023         (keep (fn [[symbol {:keys [decl]}]]
024                 (when (= :fn (:op decl))
025                   symbol)))
026         set))
027  
028  (defn- imported-airj-symbols
029    [module]
030    (->> (:imports module)
031         (filter #(= :airj-import (:op %)))
032         (mapcat (fn [import]
033                   (map (fn [symbol]
034                          [symbol (:module import)])
035                        (:symbols import))))))
036  
037  (defn- check-imports
038    [module]
039    (let [declared (declared-symbols module)]
040      (reduce (fn [seen [symbol imported-module]]
041                (when (contains? declared symbol)
042                  (fail! "Imported symbol conflicts with local declaration."
043                         {:module (:name module)
044                          :symbol symbol
045                          :imported-module imported-module}))
046                (when-let [prior-module (get seen symbol)]
047                  (fail! "Duplicate imported symbol."
048                         {:module (:name module)
049                          :symbol symbol
050                          :imported-module imported-module
051                          :prior-module prior-module}))
052                (assoc seen symbol imported-module))
053              {}
054              (imported-airj-symbols module))))
055  
056  (defn- check-exports
057    [module]
058    (let [declared (declared-symbols module)]
059      (doseq [export (:exports module)]
060        (when-not (contains? declared export)
061          (fail! "Unresolved export."
062                 {:module (:name module)
063                  :export export})))))
064  
065  (defn- union-pattern-binders
066    [pattern]
067    (apply set/union #{} (map pattern-binders (:args pattern))))
068  
069  (defn- record-pattern-binders
070    [pattern]
071    (apply set/union #{} (map (comp pattern-binders :pattern) (:fields pattern))))
072  
073  (def ^:private pattern-binder-handlers
074    {:wildcard-pattern (constantly #{})
075     :literal-pattern (constantly #{})
076     :binder-pattern (fn [pattern] #{(:name pattern)})
077     :union-pattern union-pattern-binders
078     :record-pattern record-pattern-binders})
079  
080  (defn pattern-binders
081    [pattern]
082    (if-let [handler (get pattern-binder-handlers (:op pattern))]
083      (handler pattern)
084      #{}))
085  
086  (defn- resolve-case
087    [case env walk]
088    (let [binder-names (pattern-binders (:pattern case))]
089      (walk (:body case) (set/union env binder-names))))
090  
091  (defn- resolve-expr-seq
092    [exprs env walk]
093    (reduce (fn [current-env expr]
094              (walk expr current-env)
095              (if (= :var (:op expr))
096                (conj current-env (:name expr))
097                current-env))
098            env
099            exprs)
100    nil)
101  
102  (defn- resolve-local-expr
103    [expr env _walk]
104    (when-not (contains? env (:name expr))
105      (fail! "Unresolved local."
106             {:name (:name expr)})))
107  
108  (defn- resolve-call-expr
109    [expr env walk]
110    (walk (:callee expr) env)
111    (resolve-expr-seq (:args expr) env walk))
112  
113  (defn- resolve-construct-expr
114    [expr env walk]
115    (resolve-expr-seq (:args expr) env walk))
116  
117  (defn- resolve-record-get-expr
118    [expr env walk]
119    (walk (:target expr) env))
120  
121  (defn- resolve-if-expr
122    [expr env walk]
123    (walk (:test expr) env)
124    (walk (:then expr) env)
125    (walk (:else expr) env))
126  
127  (defn- resolve-match-expr
128    [expr env walk]
129    (walk (:target expr) env)
130    (doseq [case (:cases expr)]
131      (resolve-case case env walk)))
132  
133  (defn- resolve-scoped-bindings
134    [bindings env walk]
135    (reduce (fn [names binding]
136              (do
137                (walk (:expr binding) (set/union env names))
138                (conj names (:name binding))))
139            #{}
140            bindings))
141  
142  (defn- resolve-let-expr
143    [expr env walk]
144    (let [binding-names (resolve-scoped-bindings (:bindings expr) env walk)]
145      (walk (:body expr) (set/union env binding-names))))
146  
147  (defn- resolve-lambda-expr
148    [expr env walk]
149    (walk (:body expr)
150          (set/union env (set (map :name (:params expr))))))
151  
152  (defn- resolve-try-expr
153    [expr env walk]
154    (walk (:body expr) env)
155    (doseq [catch (:catches expr)]
156      (walk (:body catch) (conj env (:name catch))))
157    (when-let [finally (:finally expr)]
158      (walk finally env)))
159  
160  (defn- resolve-var-expr
161    [expr env walk]
162    (walk (:init expr) env))
163  
164  (defn- resolve-set-expr
165    [expr env walk]
166    (walk (:expr expr) env))
167  
168  (def ^:private expr-resolvers
169    {:local resolve-local-expr
170     :call resolve-call-expr
171     :construct resolve-construct-expr
172     :variant resolve-construct-expr
173     :record-get resolve-record-get-expr
174     :if resolve-if-expr
175     :match resolve-match-expr
176     :let resolve-let-expr
177     :seq (fn [expr env walk] (resolve-expr-seq (:exprs expr) env walk))
178     :lambda resolve-lambda-expr
179     :try resolve-try-expr
180     :var resolve-var-expr
181     :set resolve-set-expr
182     :loop resolve-let-expr
183     :recur (fn [expr env walk] (resolve-expr-seq (:args expr) env walk))
184     :raise (fn [expr env walk] (walk (:expr expr) env))
185     :java-new (fn [expr env walk] (resolve-expr-seq (:args expr) env walk))
186     :java-call (fn [expr env walk]
187                  (walk (:target expr) env)
188                  (resolve-expr-seq (:args expr) env walk))
189     :java-static-call (fn [expr env walk] (resolve-expr-seq (:args expr) env walk))
190     :java-get-field (fn [expr env walk] (walk (:target expr) env))
191     :java-set-field (fn [expr env walk]
192                       (walk (:target expr) env)
193                       (walk (:expr expr) env))})
194  
195  (defn resolve-expr
196    [expr env]
197    (expr-walker/walk-expr expr env expr-resolvers))
198  
199  (defn- resolve-fn-decl
200    [decl module-fns]
201    (resolve-expr (:body decl)
202                  (set/union module-fns
203                             (set (map :name (:params decl))))))
204  
205  (defn resolve-module
206    [module]
207    (let [module-fns (set/union (declared-function-symbols module)
208                                (imported-function-symbols module))]
209      (check-imports module)
210      (check-exports module)
211      (doseq [decl (:decls module)]
212        (when (= :fn (:op decl))
213          (resolve-fn-decl decl module-fns))))
214    module)
