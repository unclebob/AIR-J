211    [command options remaining-args stdin]
212    (let [[source-option output-dir] remaining-args]

224  (defn- command-result
225    [command args stdin]
226    (let [[options remaining-args] (parse-options args)]
227      (cond
228        (:project-sources options)
229        (project-result command options remaining-args)
230  
231        (:project-dir options)
232        (project-dir-result command options remaining-args)
233  
234        (contains? source-commands command)
235        (source-result command options remaining-args stdin)
236  
237        (#{"build" "run"} command)
238        (source-result command options remaining-args stdin)
239  
240        :else
241        (throw (ex-info "Unsupported command."
242                        {:args (cons command args)})))))

245    [args stdin]

249  (defn- format-project-error
250    [data]
251    (let [prefix (str "Project command failed: "
252                      (:command data)
253                      " "
254                      (:root-module data))]
255      (cond
256        (:cycle data)
257        (str prefix "\nImport cycle: " (pr-str (:cycle data)))
258  
259        (:module data)
260        (str prefix "\nMissing module: " (:module data))
261  
262        :else
