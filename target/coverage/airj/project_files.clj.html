001  (ns airj.project-files
002    (:require [airj.project-graph :as project-graph]
003              [airj.project-sources :as project-sources]
004              [clojure.java.io :as io]
005              [clojure.string :as str]))
006  
007  (defn- airj-file?
008    [file]
009    (and (.isFile file)
010         (str/ends-with? (.getName file) ".airj")))
011  
012  (defn source-map
013    [project-dir]
014    (->> (file-seq (io/file project-dir))
015         (filter airj-file?)
016         (map (fn [file]
017                [nil (slurp file)]))
018         project-sources/source-map))
019  
020  (defn reachable-source-map
021    [project-dir root-module-name]
022    (project-graph/reachable-source-map (source-map project-dir) root-module-name))
023  
024  (defn root-source
025    [project-dir root-module-name]
026    (project-sources/root-source (reachable-source-map project-dir root-module-name)
027                                 root-module-name))
028  
029  (defn compiler-options
030    [project-dir root-module-name]
031    (project-sources/compiler-options (reachable-source-map project-dir root-module-name)
032                                      root-module-name))
